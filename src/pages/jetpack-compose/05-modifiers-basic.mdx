# Basic: Modifier (padding, size, click events)

Modifier is the key to customizing Compose UI elements. It handles layout, appearance, and behavior through a chainable API.

## Basic Modifier Chaining

```kotlin
@Composable
fun BasicModifier() {
    Text(
        text = "Hello World",
        modifier = Modifier
            .fillMaxWidth()
            .padding(16.dp)
            .background(Color.Blue)
            .padding(8.dp) // Inner padding
    )
}
```

Order matters: modifiers are applied from top to bottom in the chain.

## Size Modifiers

```kotlin
@Composable
fun SizeExamples() {
    Column {
        // Fixed size
        Box(
            modifier = Modifier
                .size(100.dp)
                .background(Color.Red)
        )
        
        // Width and height separately
        Box(
            modifier = Modifier
                .width(200.dp)
                .height(50.dp)
                .background(Color.Green)
        )
        
        // Fill available space
        Box(
            modifier = Modifier
                .fillMaxWidth()
                .height(60.dp)
                .background(Color.Blue)
        )
        
        // Fill with fraction
        Box(
            modifier = Modifier
                .fillMaxWidth(0.5f) // 50% width
                .height(40.dp)
                .background(Color.Yellow)
        )
    }
}
```

Use `fillMaxSize()`, `fillMaxWidth()`, `fillMaxHeight()` for responsive layouts.

## Padding Modifiers

```kotlin
@Composable
fun PaddingExamples() {
    Column {
        // Uniform padding
        Text(
            "Uniform padding",
            modifier = Modifier
                .background(Color.LightGray)
                .padding(16.dp)
        )
        
        // Directional padding
        Text(
            "Directional padding",
            modifier = Modifier
                .background(Color.LightGray)
                .padding(
                    start = 16.dp,
                    top = 8.dp,
                    end = 16.dp,
                    bottom = 24.dp
                )
        )
        
        // Horizontal and vertical
        Text(
            "H/V padding",
            modifier = Modifier
                .background(Color.LightGray)
                .padding(horizontal = 20.dp, vertical = 12.dp)
        )
        
        // Using PaddingValues
        Text(
            "PaddingValues",
            modifier = Modifier
                .background(Color.LightGray)
                .padding(PaddingValues(16.dp))
        )
    }
}
```

Padding outside background vs inside affects visual appearance.

## Click Events

```kotlin
@Composable
fun ClickExamples() {
    var clickCount by remember { mutableStateOf(0) }
    
    Column(
        modifier = Modifier.padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        // Basic click
        Text(
            text = "Clicked $clickCount times",
            modifier = Modifier
                .clickable { clickCount++ }
                .padding(16.dp)
                .background(Color.LightBlue)
        )
        
        // Click with ripple effect
        Text(
            text = "Click with ripple",
            modifier = Modifier
                .clickable(
                    indication = rememberRipple(),
                    interactionSource = remember { MutableInteractionSource() }
                ) { clickCount++ }
                .padding(16.dp)
                .background(Color.LightGreen)
        )
        
        // No ripple effect
        Text(
            text = "No ripple",
            modifier = Modifier
                .clickable(
                    indication = null,
                    interactionSource = remember { MutableInteractionSource() }
                ) { clickCount++ }
                .padding(16.dp)
                .background(Color.LightCoral)
        )
    }
}
```

`clickable` adds touch feedback and accessibility support automatically.

## Advanced Click Handling

```kotlin
@Composable
fun AdvancedClickHandling() {
    var message by remember { mutableStateOf("No interaction") }
    
    Box(
        modifier = Modifier
            .size(200.dp)
            .background(Color.Gray)
            .pointerInput(Unit) {
                detectTapGestures(
                    onTap = { message = "Tap detected" },
                    onLongPress = { message = "Long press detected" },
                    onDoubleTap = { message = "Double tap detected" }
                )
            },
        contentAlignment = Alignment.Center
    ) {
        Text(message, color = Color.White)
    }
}
```

Use `pointerInput` for complex gesture detection beyond simple clicks.

## Conditional Modifiers

```kotlin
@Composable
fun ConditionalModifiers() {
    var isSelected by remember { mutableStateOf(false) }
    
    Text(
        text = "Toggle me",
        modifier = Modifier
            .clickable { isSelected = !isSelected }
            .padding(16.dp)
            .then(
                if (isSelected) {
                    Modifier.background(Color.Blue)
                } else {
                    Modifier.background(Color.Gray)
                }
            )
            .padding(8.dp)
    )
}
```

Use `then()` to conditionally apply modifiers based on state.

## Common Size and Layout Patterns

```kotlin
@Composable
fun CommonPatterns() {
    Column {
        // Square aspect ratio
        Box(
            modifier = Modifier
                .fillMaxWidth()
                .aspectRatio(1f)
                .background(Color.Red)
        )
        
        // Minimum size constraints
        Text(
            "Min size",
            modifier = Modifier
                .defaultMinSize(minWidth = 100.dp, minHeight = 48.dp)
                .background(Color.Green)
                .padding(8.dp)
        )
        
        // Weight in Column/Row
        Row(modifier = Modifier.fillMaxWidth()) {
            Box(
                modifier = Modifier
                    .weight(1f)
                    .height(50.dp)
                    .background(Color.Blue)
            )
            Box(
                modifier = Modifier
                    .weight(2f) // Takes 2x space
                    .height(50.dp)
                    .background(Color.Yellow)
            )
        }
    }
}
```

`weight()` distributes space proportionally in Row/Column layouts.

## Combining Everything

```kotlin
@Composable
fun CombinedExample() {
    var count by remember { mutableStateOf(0) }
    var isLongPressed by remember { mutableStateOf(false) }
    
    Box(
        modifier = Modifier
            .fillMaxWidth()
            .padding(16.dp)
            .background(
                if (isLongPressed) Color.Red else Color.Blue,
                RoundedCornerShape(8.dp)
            )
            .clickable { count++ }
            .pointerInput(Unit) {
                detectTapGestures(
                    onLongPress = { 
                        isLongPressed = true 
                    },
                    onTap = { 
                        isLongPressed = false 
                    }
                )
            }
            .padding(24.dp)
            .defaultMinSize(minHeight = 80.dp),
        contentAlignment = Alignment.Center
    ) {
        Text(
            text = "Count: $count\n${if (isLongPressed) "Long pressed!" else "Tap to count"}",
            color = Color.White,
            textAlign = TextAlign.Center
        )
    }
}
```

## Key Principles

- **Order matters**: modifiers apply sequentially from top to bottom
- **Padding placement**: before background = outer padding, after = inner padding
- **Size constraints**: use `fillMax*` for responsive, fixed `size()` for absolute
- **Click handling**: `clickable` for simple, `pointerInput` for complex gestures
- **State-driven**: use conditional modifiers with `then()` for dynamic behavior
- **Performance**: remember interaction sources and ripple instances
