# Scaffold

## Overview

Scaffold is a high-level composable that provides the basic structure for a screen in Material Design. It implements the basic material layout structure with slots for common UI elements like app bars, floating action buttons, drawers, and snackbars.

## Setup

Scaffold is part of the Material3 library:

```kotlin
implementation "androidx.compose.material3:material3:1.1.2"
```

## Basic Usage

### Simple Scaffold

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun BasicScaffold() {
    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("Basic Scaffold") }
            )
        }
    ) { innerPadding ->
        LazyColumn(
            modifier = Modifier.padding(innerPadding)
        ) {
            items(100) { index ->
                Text(
                    text = "Item $index",
                    modifier = Modifier.padding(16.dp)
                )
            }
        }
    }
}
```

Creates a basic screen with a top app bar and scrollable content. The `innerPadding` ensures content doesn't overlap with system bars.

```kotlin
fun ModalDrawerExample() {
    val drawerState = rememberDrawerState(DrawerValue.Closed)

                        icon = { Icon(icons[index], contentDescription = null) },

                        onClick = {
                            scope.launch { drawerState.close() }
                            selectedItem = index
                        },
                        modifier = Modifier.padding(NavigationDrawerItemDefaults.ItemPadding)
                    )
                }

        Scaffold(
            topBar = {
                TopAppBar(
                    title = { Text(menuItems[selectedItem]) },
                    navigationIcon = {
                        IconButton(
                            onClick = {
                                scope.launch {

                                    }
                                }
                            }
                        ) {
                            Icon(Icons.Default.Menu, contentDescription = "Menu")
                        }
                    }
                )
            }
        ) { innerPadding ->
            Box(
                modifier = Modifier
                    .fillMaxSize()
                    .padding(innerPadding),
                contentAlignment = Alignment.Center
            ) {
                Text("${menuItems[selectedItem]} Content")
            }
        }
    }
}
```

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun CompleteAppStructure() {
    var selectedTab by remember { mutableIntStateOf(0) }
    var fabVisible by remember { mutableStateOf(true) }
    val snackbarHostState = remember { SnackbarHostState() }
    val scope = rememberCoroutineScope()
    
    val tabs = listOf("Home", "Explore", "Library")
    val tabIcons = listOf(Icons.Default.Home, Icons.Default.Search, Icons.Default.List)
    
    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text(tabs[selectedTab]) },
                actions = {
                    IconButton(onClick = { fabVisible = !fabVisible }) {
                        Icon(
                            if (fabVisible) Icons.Default.VisibilityOff else Icons.Default.Visibility,
                            contentDescription = "Toggle FAB"
                        )
                    }
                }
            )
        },
        bottomBar = {
            NavigationBar {
                tabs.forEachIndexed { index, tab ->
                    NavigationBarItem(
                        icon = { Icon(tabIcons[index], contentDescription = tab) },
                        label = { Text(tab) },
                        selected = selectedTab == index,
                        onClick = { selectedTab = index }
                    )
                }
            }
        },
        floatingActionButton = {
            AnimatedVisibility(
                visible = fabVisible,
                enter = scaleIn(),
                exit = scaleOut()
            ) {
                FloatingActionButton(
                    onClick = {
                        scope.launch {
                            snackbarHostState.showSnackbar("Action performed!")
                        }
                    }
                ) {
                    Icon(Icons.Default.Add, contentDescription = "Add")
                }
            }
        },
        snackbarHost = { SnackbarHost(snackbarHostState) }
    ) { innerPadding ->
        when (selectedTab) {
            0 -> HomeContent(innerPadding)
            1 -> ExploreContent(innerPadding)
            2 -> LibraryContent(innerPadding)
        }
    }
}
```

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun SearchAppBarExample() {
    var searchActive by remember { mutableStateOf(false) }
    var searchQuery by remember { mutableStateOf("") }
    val items = remember { (1..100).map { "Item $it" } }
    val filteredItems = remember(searchQuery) {
        if (searchQuery.isEmpty()) items else items.filter { 
            it.contains(searchQuery, ignoreCase = true) 
        }
    }
    
    Scaffold(
        topBar = {
            if (searchActive) {
                SearchBar(
                    query = searchQuery,
                    onQueryChange = { searchQuery = it },
                    onSearch = { searchActive = false },
                    active = searchActive,
                    onActiveChange = { searchActive = it },
                    placeholder = { Text("Search items...") },
                    leadingIcon = {
                        IconButton(onClick = { 
                            searchActive = false
                            searchQuery = ""
                        }) {
                            Icon(Icons.Default.ArrowBack, contentDescription = "Back")
                        }
                    },
                    trailingIcon = {
                        if (searchQuery.isNotEmpty()) {
                            IconButton(onClick = { searchQuery = "" }) {
                                Icon(Icons.Default.Clear, contentDescription = "Clear")
                            }
                        }
                    }
                ) {
                    LazyColumn {
                        items(filteredItems) { item ->
                            ListItem(
                                headlineContent = { Text(item) },
                                modifier = Modifier.clickable {
                                    searchQuery = item
                                    searchActive = false
                                }
                            )
                        }
                    }
                }
            } else {
                TopAppBar(
                    title = { Text("Search Example") },
                    actions = {
                        IconButton(onClick = { searchActive = true }) {
                            Icon(Icons.Default.Search, contentDescription = "Search")
                        }
                    }
                )
            }
        }
    ) { innerPadding ->
        LazyColumn(
            modifier = Modifier.padding(innerPadding)
        ) {
            items(filteredItems) { item ->
                ListItem(
                    headlineContent = { Text(item) }
                )
                HorizontalDivider()
            }
        }
    }
}
```

### 1. State Management

```kotlin
    val scope = rememberCoroutineScope()
```

## Best Practices

1. **Always use innerPadding** from Scaffold's content lambda to avoid UI overlap
2. **Remember state objects** like `SnackbarHostState` and `DrawerState`
3. **Use appropriate WindowInsets** for edge-to-edge layouts
4. **Animate FAB visibility** based on scroll state for better UX
5. **Handle back press** properly with drawers and search bars
6. **Use proper content padding** for lists and grids
7. **Test on different screen sizes** and orientations

## Common Patterns

### 1. Master-Detail Layout

```kotlin
@Composable
fun MasterDetailScaffold() {
    var selectedItem by remember { mutableStateOf<String?>(null) }
    val isWideScreen = LocalConfiguration.current.screenWidthDp > 600
    
    if (isWideScreen) {
        Row {
            // Master pane
            Surface(
                modifier = Modifier.weight(1f),
                tonalElevation = 1.dp
            ) {
                MasterPane(
                    onItemSelected = { selectedItem = it },
                    selectedItem = selectedItem
                )
            }
            
            // Detail pane
            Surface(modifier = Modifier.weight(2f)) {
                DetailPane(selectedItem)
            }
        }
    } else {
        // Single pane with navigation
        if (selectedItem == null) {
            Scaffold(
                topBar = {
                    TopAppBar(title = { Text("Items") })
                }
            ) { innerPadding ->
                MasterPane(
                    onItemSelected = { selectedItem = it },
                    selectedItem = selectedItem,
                    modifier = Modifier.padding(innerPadding)
                )
            }
        } else {
            Scaffold(
                topBar = {
                    TopAppBar(
                        title = { Text(selectedItem ?: "") },
                        navigationIcon = {
                            IconButton(onClick = { selectedItem = null }) {
                                Icon(Icons.Default.ArrowBack, contentDescription = "Back")
                            }
                        }
                    )
                }
            ) { innerPadding ->
                DetailPane(
                    selectedItem,
                    modifier = Modifier.padding(innerPadding)
                )
            }
        }
    }
}

@Composable
fun MasterPane(
    onItemSelected: (String) -> Unit,
    selectedItem: String?,
    modifier: Modifier = Modifier
) {
    LazyColumn(modifier = modifier) {
        items(20) { index ->
            val item = "Item $index"
            ListItem(
                headlineContent = { Text(item) },
                modifier = Modifier.clickable { onItemSelected(item) },
                colors = if (item == selectedItem) {
                    ListItemDefaults.colors(containerColor = MaterialTheme.colorScheme.primaryContainer)
                } else {
                    ListItemDefaults.colors()
                }
            )
        }
    }
}

@Composable
fun DetailPane(
```
